#!/bin/bash

# Arreglos para almacenar los números ingresados
numeros=()
mayor=0
menor=0

# Solicitar 10 números al usuario
for ((i=1; i<=10; i++)); do
  echo "Introduce el número $i:"
  read numero

  # Validar que se ingrese un número entero
  if ! [[ "$numero" =~ ^[0-9]+$ ]]; then
    echo "Error: Debes ingresar un número entero."
    exit 1
  fi

  # Almacenar el número ingresado en el arreglo
  numeros+=($numero)

  # Actualizar el mayor y menor número
  if ((i == 1)); then
    mayor=$numero
    menor=$numero
  else
    if ((numero > mayor)); then
      mayor=$numero
    fi
    if ((numero < menor)); then
      menor=$numero
    fi
  fi
done

# Mostrar el mayor y menor número ingresado
echo "El mayor número ingresado es: $mayor"
echo "El menor número ingresado es: $menor"
